---- node ----------------------------------------------------
// 1. node 버전확인
$ node -v
v6.0.0

// 2. 기존에 사용 했을수도 있는 npm cache값을 제거해줍니다.
$ npm cache clean -f

// 3. n 모듈 설치
// (n 은 Node.js 버전을 관리해주는 플러그인)
$ npm install -g n

// 4. node 안정버전 설치
$ n stable

// 4. lts버전 설치
$ n lts

// 4. 특정버전 설치 ex> $ n <version> 
$ n 10.0.0

// 버전삭제 
$ n rm 10.0.0

---- npm ----------------------------------------------------
// 1. npm 버전 확인
$ npm -v
6.9.0

// 2. npm 재설치
$ sudo npm install -g npm
